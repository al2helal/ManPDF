\hyperdef{}{NAME}{\section{\hyperref[NAME]{NAME}}\label{NAME}}

asn1\_decode\_simple\_der - API function

\hyperdef{}{SYNOPSIS}{\section{\hyperref[SYNOPSIS]{SYNOPSIS}}\label{SYNOPSIS}}

\textbf{\#include \textless{}libtasn1.h\textgreater{}}

~

\textbf{int asn1\_decode\_simple\_der(unsigned int}\emph{etype}\textbf{,
const unsigned char *}\emph{der}\textbf{, unsigned
int}\emph{der\_len}\textbf{, const unsigned char **}\emph{str}\textbf{,
unsigned int *}\emph{str\_len}\textbf{);}

\hyperdef{}{ARGUMENTS}{\section{\hyperref[ARGUMENTS]{ARGUMENTS}}\label{ARGUMENTS}}

\begin{description}
\itemsep1pt\parskip0pt\parsep0pt
\item[unsigned int etype]
The type of the string to be encoded (ASN1\_ETYPE\_)
\end{description}

\begin{description}
\itemsep1pt\parskip0pt\parsep0pt
\item[const unsigned char * der]
the encoded string
\end{description}

\begin{description}
\itemsep1pt\parskip0pt\parsep0pt
\item[unsigned int der\_len]
the bytes of the encoded string
\end{description}

\begin{description}
\itemsep1pt\parskip0pt\parsep0pt
\item[const unsigned char ** str]
a pointer to the data
\end{description}

\begin{description}
\itemsep1pt\parskip0pt\parsep0pt
\item[unsigned int * str\_len]
the length of the data
\end{description}

\hyperdef{}{DESCRIPTION}{\section{\hyperref[DESCRIPTION]{DESCRIPTION}}\label{DESCRIPTION}}

Decodes a simple DER encoded type (e.g. a string, which is not
constructed). The output is a pointer inside the \emph{der} .

\hyperdef{}{RETURNS}{\section{\hyperref[RETURNS]{RETURNS}}\label{RETURNS}}

\textbf{ASN1\_SUCCESS} if successful or an error value.
